rails new cloudsearch-rails-demo

First, add the dependencies on asari. Your gemfile should look like this:

{% highlight ruby %}
source 'https://rubygems.org'
gem 'rails', '4.0.2'
gem 'sqlite3'
gem 'sass-rails', '~> 4.0.0'
gem 'uglifier', '>= 1.3.0'
gem 'coffee-rails', '~> 4.0.0'
gem 'jquery-rails'
gem 'turbolinks'
gem 'jbuilder', '~> 1.2'

gem 'asari'
{% endhighlight %}

bundle install

rails g scaffold Post body:text category:string

rake db:migrate
rails s
